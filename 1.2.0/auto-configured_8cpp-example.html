#include "com/foo/config.h"
auto rootLogger = com::foo::getLogger();
struct ExampleStaticData {
ExampleStaticData() {
LOG4CXX_DEBUG(rootLogger, "static initializer message");
}
} static_object;
int main() {
LOG4CXX_INFO(rootLogger, "main function message");
return EXIT_SUCCESS;
}
